        function openCalendly(type) {
            let url = '';
            
            switch(type) {
                case 'hudanalyse':
                    url = 'https://calendly.com/your-calendly/hudanalyse';
                    break;
                case 'microneedling':
                    url = 'https://calendly.com/your-calendly/microneedling';
                    break;
                case 'holistisk':
                    url = 'https://calendly.com/your-calendly/holistisk';
                    break;
                default:
                    url = 'https://calendly.com/your-calendly';
            }
            
            document.getElementById('calendly-widget').innerHTML = '';
            Calendly.initInlineWidget({
                url: url,
                parentElement: document.getElementById('calendly-widget'),
                prefill: {},
                utm: {}
            });
            
            // Scroll to calendly section
            document.getElementById('calendly').scrollIntoView({behavior: 'smooth'});
        }
